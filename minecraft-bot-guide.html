Minecraft Bot Command Guide

1. Update and Upgrade:

pkg update && pkg upgrade

2. Install Node.js:

pkg install nodejs-lts

3. Check Node.js and npm Version:

node -v
npm -v

4. Install Mineflayer:

npm install mineflayer

5. Create and Navigate to Minecraft Bot Folder:

mkdir minecraft-bot
cd minecraft-bot

6. Create bot.js:

nano bot.js

7. Paste this Code in bot.js:

const mineflayer = require('mineflayer');

const botUsername = 'bot';
const botPassword = 'pitaji';

function createBot() {
  const bot = mineflayer.createBot({
    host: 'angrymc.aternos.me',
    port: 13331,
    username: botUsername,
    password: botPassword,
    keepAlive: true,
    version: '1.20.4'
  });

  bot.on('spawn', () => {
    bot.look(0, 0, true);
    bot.chat(`/register ${botPassword} ${botPassword}`);
    setTimeout(() => {
      bot.chat(`/login ${botPassword}`);
    }, 3000);

    setInterval(() => {
      bot.setControlState('jump', true);
      setTimeout(() => {
        bot.setControlState('jump', false);
      }, Math.floor(Math.random() * 700) + 300);
    }, Math.floor(Math.random() * 5000) + 5000);
  });

  bot.on('end', () => setTimeout(createBot, 5000));
  bot.on('error', () => setTimeout(createBot, 5000));
  bot.on('kicked', () => setTimeout(createBot, 5000));
}

createBot();

8. Install pm2:

npm install -g pm2

9. Start the Bot:

pm2 start bot.js --name "MinecraftBot"

10. Save pm2 Configuration:

pm2 save
